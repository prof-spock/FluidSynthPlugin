121INLINE
122AudioSample AudioSampleFifoQueue::pop ()
123{
124 Assertion_pre(_length > 0, "cannot pop from empty queue");
125 AudioSample result = _data.at(_firstIndex++);
126 _length--;
127
128 /* when firstIndex exceeds a certain boundary (here 10% of
129 * capacity), shift data to the left */
130 if (_firstIndex > _capacity / 10) {
131 for (Natural i = 0; i < _length; i++) {
132 _data[i] = _data[_firstIndex + i];
133 }
134
135 _firstIndex = 0;
136 }
137
138 return result;
139}

143INLINE
144void AudioSampleFifoQueue::pop (INOUT AudioSampleList& sampleList,
145 IN Natural position,
146 IN Natural count)
147{
148 const Natural listLength = sampleList.length();
149 Assertion_pre(position + count <= listLength,
150 "count exceeds list length");
151 Assertion_pre(count <= length(),
152 "count exceeds queue length");
153
154 Natural j = position;
155
156 for (Natural i = 0; i < count; i++) {
157 sampleList[j++] = pop();
158 }
159}
