93 {
94 Assertion_pre(base >= 2 && base <= 36,
95 "base must be between 2 and 36");
96 String result = "";

98 Natural length = 0;
99 String digitList = "0123456789ABCDEFGHIJKLMNOPQRSTUVWXYZ";
100
101 while (true) {
102 Natural digit = remainingValue % base;
103 remainingValue = remainingValue / base;
104 result = digitList[(size_t) digit] + result;
105 length++;
106
107 if (remainingValue == 0) {
108 break;
109 }
110 }
111
112 while (length < precision) {
113 result = padString + result;
114 length++;
115 }

118 }
