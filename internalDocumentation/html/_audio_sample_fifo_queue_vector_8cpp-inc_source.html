46{

48 const Natural channelCount = queueCount();
49 const Natural sampleCount = Natural::minimum(audioFrameCount,

52 (isGroupedByFrames ? audioFrameCount : channelCount);
53 Natural groupLength =
54 (isGroupedByFrames ? channelCount : audioFrameCount);
55
56 for (Natural i = 0; i < groupCount; i++) {
57 String groupString = "(";

60 const AudioSample& sample = (isGroupedByFrames

62 : at(i).at(j));
63 groupString += (j == 0 ? "" : ", ") + TOSTRING(sample);
64 }
65
66 groupString += (sampleCount < audioFrameCount ? ", ..." : "");
67 groupString += ")";
68 result += (result == "" ? "" : ", ") + groupString;
69 }

71 result = STR::expand("AudioSampleFifoQueueVector(%1)", result);
